CREATE TABLE Course(
    departmentName     TEXT,
	  subNumber		       TEXT,
   	name               TEXT,  
   	description        TEXT,
   	gradeBasis         TEXT,
   	maxCreditHours     TEXT,
   	minCreditHours     TEXT,
   	maxLectureHours    TEXT,
   	minLectureHours    TEXT,
   	maxLabHours        TEXT,
   	minLabHours        TEXT,
   	dept               TEXT,
   	sectionsLink       TEXT,
   	resctrictions      TEXT,
   	prerequisites      TEXT,
    PRIMARY KEY (departmentName, subNumber)
);


CREATE TABLE User(
	userName      TEXT      PRIMARY KEY,
	password      TEXT
);


CREATE TABLE Works_On(
     userName		              TEXT,
     departmentName           TEXT,
     subNumber          TEXT,

     FOREIGN KEY (userName) 
     	REFERENCES User(userName)
     	ON DELETE CASCADE
        ON UPDATE CASCADE
     ,

     FOREIGN KEY (departmentName) 
      REFERENCES Course(departmentName)
      ON DELETE CASCADE
        ON UPDATE CASCADE
     ,

     FOREIGN KEY (subNumber) 
     	REFERENCES Course(subNumber)
     	ON DELETE CASCADE
        ON UPDATE CASCADE
     ,
     PRIMARY KEY (userName, departmentName , subNumber)
);
