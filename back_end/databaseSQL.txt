CREATE TABLE Course(
	subNumber		   TEXT PRIMARY KEY,
   	name               TEXT,  
   	description        TEXT,
   	gradeBasis         TEXT,
   	maxCreditHours     TEXT,
   	minCreditHours     TEXT,
   	maxLectureHours    TEXT,
   	minLectureHours    TEXT,
   	maxLabHours        TEXT,
   	minLabHours        TEXT,
   	dept               TEXT,
   	sectionsLink       TEXT,
   	resctrictions      TEXT,
   	prerequisites      TEXT
);



CREATE TABLE User(
	userName      TEXT      PRIMARY KEY,
	password      TEXT,
);


CREATE TABLE Works_On(
     userName		          TEXT,
     courseSubNumber          TEXT,
     FOREIGN KEY (userName) 
     	REFERENCES User(userName)
     	ON DELETE CASCADE
        ON UPDATE CASCADE
     ,
     FOREIGN KEY (courseSubNumber) 
     	REFERENCES Course(subNumber)
     	ON DELETE CASCADE
        ON UPDATE CASCADE
     ,
     PRIMARY KEY (userName, courseSubNumber)
);
